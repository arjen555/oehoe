function numToNL(n){
  if(n===0)return'nul';
  var neg=n<0;n=Math.abs(n);
  var ones=['','een','twee','drie','vier','vijf','zes','zeven','acht','negen','tien','elf','twaalf','dertien','veertien','vijftien','zestien','zeventien','achttien','negentien'];
  var tens=['','','twintig','dertig','veertig','vijftig','zestig','zeventig','tachtig','negentig'];
  function b100(n){if(n<20)return ones[n];var t=Math.floor(n/10),u=n%10;return u>0?ones[u]+'en'+tens[t]:tens[t];}
  function b1000(n){if(n<100)return b100(n);var h=Math.floor(n/100),r=n%100;return(h===1?'honderd':ones[h]+'honderd')+(r===0?'':( r<10?' en ':'')+b100(r));}
  var res='';
  if(n>=1000000){var m=Math.floor(n/1000000);res+=b1000(m)+(m===1?' miljoen':' miljoen ');n=n%1000000;}
  if(n>=1000){var k=Math.floor(n/1000);res+=(k===1?'duizend':b1000(k)+'duizend');n=n%1000;if(n>0)res+=' ';}
  if(n>0)res+=b1000(n);
  return(neg?'min ':'')+res.trim();
}

function rand(min,max){return Math.floor(Math.random()*(max-min+1))+min;}
function generateProblem(){
  var p1max=cfg.getal1Place*10-1,p1min=Math.max(1,cfg.getal1Place);
  var p2max=cfg.getal2Place*10-1,p2min=Math.max(1,cfg.getal2Place);
  if(cfg.op==='optellen'){var a=rand(p1min,p1max),b=rand(p2min,p2max);return{a:a,b:b,op:'+',answer:a+b};}
  else if(cfg.op==='aftrekken'){
    var a=rand(p1min,p1max),b=rand(p2min,Math.min(p2max,a-1));
    if(b<p2min)b=p2min;if(a<=b)a=b+rand(1,10);
    return{a:a,b:b,op:'-',answer:a-b};
  } else {var a=rand(p1min,p1max),b=rand(p2min,p2max);return{a:a,b:b,op:'*',answer:a*b};}
}

function firstRealInRow(targetRow){
  if(cfg.method==='kolomsgewijs'&&cfg.op==='aftrekken'&&cfg.hulp&&currentProblem){
    var p=currentProblem;
    var aStr=String(p.a),bStr=String(p.b);
    var maxPlaces=Math.max(aStr.length,bStr.length);
    var pos=maxPlaces-1-targetRow;
    if(pos>=0){
      var pv=Math.pow(10,pos);
      var dA=Math.floor(p.a/pv%10)*pv;
      var dB=Math.floor(p.b/pv%10)*pv;
      var val=dA-dB;
      if(val<0){
        // Vind het eerste echte (niet-optionele) cijfervakje in deze rij
        var eersteEchteIdx=null;
        for(var j=0;j<inputFields.length;j++){
          if(rowInfo[j]===targetRow&&!isOptional(expectedAnswers[j])&&expectedAnswers[j]!==-2&&inputFields[j]&&!inputFields[j].disabled){
            eersteEchteIdx=j;break;
          }
        }
        if(eersteEchteIdx!==null){
          // Ga één stap terug in inputFields binnen dezelfde rij
          for(var j=eersteEchteIdx-1;j>=0;j--){
            if(rowInfo[j]===targetRow&&inputFields[j]&&!inputFields[j].disabled){
              return inputFields[j];
            }
          }
        }
      }
    }
  }
  for(var j=0;j<inputFields.length;j++){
    if(rowInfo[j]===targetRow&&!isOptional(expectedAnswers[j])&&inputFields[j]&&!inputFields[j].disabled)
      return inputFields[j];
  }
  for(var j=0;j<inputFields.length;j++){
    if(rowInfo[j]===targetRow&&expectedAnswers[j]!==-2&&inputFields[j]&&!inputFields[j].disabled)
      return inputFields[j];
  }
  return null;
}

function addAutoAdvance(el,ri){
  el.addEventListener('focus',function(){el.select();});
  el.addEventListener('input',function(){
    var v=el.value.replace(/[^0-9\-]/g,'');
    if(v.length>1)v=v.slice(-1);
    el.value=v;
    if(v.length===1){
      var i=inputFields.indexOf(el);
      var next=null;
      for(var j=i+1;j<inputFields.length;j++){
        if(inputFields[j]&&!inputFields[j].disabled){next=inputFields[j];break;}
      }
      if(next){
        var ni=inputFields.indexOf(next);
        var nextRow=rowInfo[ni];
        if(nextRow!==ri&&nextRow>=0){
          if(cfg.hulp){
            var real=firstRealInRow(nextRow);
            if(real)next=real;
          } else {
            for(var j=ni;j<inputFields.length;j++){
              if(rowInfo[j]===nextRow&&expectedAnswers[j]!==-2&&inputFields[j]&&!inputFields[j].disabled){next=inputFields[j];break;}
            }
          }
        } else if(isOptional(expectedAnswers[ni])&&expectedAnswers[ni]!==-2&&cfg.hulp){
          var targetRow=rowInfo[ni];
          for(var j=ni+1;j<inputFields.length;j++){
            if(rowInfo[j]===targetRow&&!isOptional(expectedAnswers[j])&&inputFields[j]&&!inputFields[j].disabled){next=inputFields[j];break;}
          }
        }
        next.focus();
      }
    }
  });
  el.addEventListener('keydown',function(e){
    if(e.key==='Enter'){checkStep();return;}
    var i=inputFields.indexOf(el);
    if(e.key==='Backspace'){e.preventDefault();el.value='';for(var j=i-1;j>=0;j--){if(inputFields[j]&&!inputFields[j].disabled){inputFields[j].focus();break;}}return;}
    if(e.key==='ArrowLeft'){e.preventDefault();for(var j=i-1;j>=0;j--){if(inputFields[j]&&!inputFields[j].disabled){inputFields[j].focus();break;}}}
    if(e.key==='ArrowRight'){e.preventDefault();for(var j=i+1;j<inputFields.length;j++){if(inputFields[j]&&!inputFields[j].disabled){inputFields[j].focus();break;}}}
    if(e.key==='ArrowUp'||e.key==='ArrowDown'){
      e.preventDefault();var curRow=rowInfo[i],curCol=0;
      for(var j=0;j<i;j++){if(rowInfo[j]===curRow)curCol++;}
      var targetRow=e.key==='ArrowUp'?curRow-1:curRow+1,colIdx=0;
      for(var j=0;j<inputFields.length;j++){if(rowInfo[j]===targetRow){if(colIdx===curCol&&inputFields[j]&&!inputFields[j].disabled){inputFields[j].focus();break;}colIdx++;}}
    }
  });
  el.addEventListener('focus',function(){
    if(cfg.hulp&&ri>=0&&ri!==currentHulpRow){
      currentHulpRow=ri;
      updateHulpHighlight(ri);
      if(cfg.veelHulp)showVeelHulpForRow(ri);
    }
  });
}

  document.querySelectorAll('.cell-input.kolom-hulp-active').forEach(function(c){c.classList.remove('kolom-hulp-active');});

      if(cell.tagName==='INPUT'){cell.classList.add('kolom-hulp-active');}
      else{cell.classList.add('hulp-active');}
    });
  }
}

  inputFields=[];rememberFields=[];expectedAnswers=[];rowInfo=[];stepHighlightCells=[];currentHulpRow=-1;aStaticCells=[];aftrDeelsomTd=null;vermEersteInput=null;vermFocusIngesteld=false;vermInputRows=[];kolomVermMeta=null;

  if(cfg.method==='kolomsgewijs'){
    if(cfg.op==='vermenigvuldigen')renderKolomVerm(p,area);
    else renderKolomOpAf(p,area);
  } else renderCijferend(p,area);
  if(!vermFocusIngesteld){
    var focusTarget=null;
    if(cfg.method==='cijferend'){
      for(var fi=inputFields.length-1;fi>=0;fi--){
        if(inputFields[fi]&&!inputFields[fi].disabled){focusTarget=inputFields[fi];break;}
      }
    } else if(cfg.method==='kolomsgewijs'&&cfg.op==='vermenigvuldigen'){
      var rij0inputs=[];
      for(var fi=0;fi<inputFields.length;fi++){
        if(rowInfo[fi]===0&&inputFields[fi]&&!inputFields[fi].disabled)rij0inputs.push(inputFields[fi]);
      }
      if(currentProblem){
        var aStr0=String(currentProblem.a),bStr0=String(currentProblem.b);
        var aDigit0=parseInt(aStr0[aStr0.length-1])||0;
        var bDigit0=parseInt(bStr0[bStr0.length-1])||0;
        var eersteVal0=aDigit0*bDigit0;
        var numCijfers0=eersteVal0===0?1:String(eersteVal0).length;
        var startPos0=rij0inputs.length-numCijfers0;
        if(startPos0<0)startPos0=0;
        focusTarget=rij0inputs[startPos0]||rij0inputs[rij0inputs.length-1]||null;
      } else {
        focusTarget=rij0inputs[rij0inputs.length-1]||null;
      }
    } else if(cfg.hulp&&cfg.method==='kolomsgewijs'&&cfg.op!=='vermenigvuldigen'){
      focusTarget=firstRealInRow(0)||null;
    } else {
      focusTarget=inputFields.find(function(el,i){return el&&!el.disabled&&expectedAnswers[i]!==-2;});
    }
    var ft=focusTarget;
    requestAnimationFrame(function(){
      requestAnimationFrame(function(){
        if(ft&&ft.isConnected){
          ft.focus();
          if(cfg.hulp){
            var ri=rowInfo[inputFields.indexOf(ft)];
            if(cfg.method==='cijferend'){
              if(cfg.op!=='*'){currentHulpRow=0;updateHulpHighlight(0);}
              // eerste deelsom wordt al uitgesproken via focus-handler
            } else {
              currentHulpRow=ri||0;updateHulpHighlight(ri||0);
            }
          }
        }
      });
    });
  }
}

      else{var dv=document.createElement('div');dv.className='cell-empty';td.appendChild(dv);}
    }
  } else {
    for(var c=0;c<numCols;c++){
      var td=r.insertCell();
      var inp=mkInput('Uitkomst cijfer '+(c+1),rowIdx);
      if(c>=startCol){expectedAnswers.push(parseInt(ansStr[c-startCol]));}
      else{expectedAnswers.push(-1);}
      inputFields.push(inp);td.appendChild(inp);
    }
  }
  area.innerHTML='';area.appendChild(t);
  if(cfg.hulp){
    var rij0v=[];
    for(var fiv=0;fiv<inputFields.length;fiv++){
      if(rowInfo[fiv]===0&&inputFields[fiv]&&!inputFields[fiv].disabled&&!isOptional(expectedAnswers[fiv]))rij0v.push(inputFields[fiv]);
    }
    var startV=rij0v[rij0v.length-1]||null;
    if(startV){setTimeout(function(){startV.focus();},0);}
  }
}

function updateAftrDeelsom(p,colIdx,cols){
  if(!cfg.veelHulp||!aftrDeelsomTd)return;
  var fromRight=cols-1-colIdx;
  var a2=String(Math.abs(p.a)),s1=cols-a2.length;
  var b2=String(Math.abs(p.b)),s2=cols-b2.length;
  var aHasDigit=colIdx>=s1;
  var bHasDigit=colIdx>=s2;
  if(!aHasDigit&&!bHasDigit){aftrDeelsomTd.innerHTML='';return;}
  var aCell=aStaticCells.find(function(x){return x.col===colIdx;});
  var bDigit=bHasDigit?parseInt(b2[colIdx-s2]):0;
  var aVal;
  if(aCell&&aCell.span){
    var html=aCell.span.innerHTML;
    if(html.indexOf('color:#FCD34D')>=0){
      var curVal=parseInt(aCell.span.dataset.cur);
      aVal=isNaN(curVal)?0:curVal;
    } else {
      var cur=aCell.span.dataset.cur!==undefined?aCell.span.dataset.cur:aCell.span.textContent;
      aVal=parseInt(cur)||0;
    }
  } else {
    aVal=aHasDigit?parseInt(a2[colIdx-s1]):0;
  }
  var kolNaam=colFullName(fromRight);
  var somTekst=aVal+' \u2212 '+bDigit;
  aftrDeelsomTd.innerHTML='<div class="cij-deelsom-wrap"><div class="cij-deelsom-kol">'+kolNaam+'</div><div class="cij-deelsom-som">'+somTekst+'</div></div>';

}

function isOptional(exp){return exp===-1||exp===-2||exp===null||exp===undefined||isNaN(exp);}
function verwachtMinteken(exp){return exp===-3;}

function checkStep(){
  if(stepChecked)return;
  var missingRequired;
  if((cfg.method==='cijferend'&&cfg.op==='*')||(cfg.method==='kolomsgewijs')){
    var maxRij=Math.max.apply(null,rowInfo.filter(function(r){return r!==undefined&&r!==null;}));
    var rijLeeg=false;
    for(var rr=0;rr<=maxRij;rr++){
      var rijInputs=inputFields.filter(function(el,i){return rowInfo[i]===rr&&el;});
      var rijHeeftWaarde=rijInputs.some(function(el){return el.value.trim()!=='';});
      if(!rijHeeftWaarde&&rijInputs.length>0){rijLeeg=true;break;}
    }
    missingRequired=rijLeeg;
  } else {
    missingRequired=inputFields.some(function(el,i){
      if(!el)return false;
      if(isOptional(expectedAnswers[i]))return false;
      if(verwachtMinteken(expectedAnswers[i]))return el.value.trim()!=='-';
      return el.value.trim()==='';
    });
  }
  if(missingRequired){showFeedback('Vul alle benodigde vakjes in voordat je controleert.','fout',null);return;}

  var onnodigGeleend=false;
  if(cfg.method==='cijferend'&&cfg.op==='-'&&currentProblem){
    var p2=currentProblem;
    var a2s=String(Math.abs(p2.a)),b2s=String(Math.abs(p2.b));
    var cols2=Math.max(a2s.length,b2s.length)+1;
    aStaticCells.forEach(function(sc){
      if(sc.span.innerHTML.indexOf('color:#FCD34D')>=0){
        var col=sc.col;
        var s1=cols2-a2s.length,s2=cols2-b2s.length;
        var origA=(col>=s1)?parseInt(a2s[col-s1]):0;
        var bDigit=(col>=s2)?parseInt(b2s[col-s2]):0;
        if(origA>=bDigit){onnodigGeleend=true;}
      }
    });
  }

  var wrongPosition=false;
  if(cfg.method==='kolomsgewijs'){
    var rows={};
    inputFields.forEach(function(el,i){
      var ri=rowInfo[i];
      if(ri<0)return;
      if(!rows[ri])rows[ri]={hasRealFilled:false,hasOptionalFilled:false};
      if(!isOptional(expectedAnswers[i])&&el.value.trim()!=='')rows[ri].hasRealFilled=true;
      if(expectedAnswers[i]===-1&&el.value.trim()!=='')rows[ri].hasOptionalFilled=true;
    });
    Object.keys(rows).forEach(function(ri){
      if(rows[ri].hasOptionalFilled&&!rows[ri].hasRealFilled)wrongPosition=true;
    });
  }

  if(wrongPosition){
    inputFields.forEach(function(el,i){
      if(!isOptional(expectedAnswers[i]))el.style.outline='3px solid #FCD34D';
    });
    showFeedback('Zet alles op de juiste positie onder elkaar.','fout',null);
    return;
  }

  inputFields.forEach(function(el){el.style.outline='';});

function showHulpAfterCheck(correctArr){
  if(!cfg.hulp||cfg.method==='cijferend')return;
  var p=currentProblem;
  var hulpArea=document.getElementById('hulp-area');hulpArea.innerHTML='';
  var msgWrap=document.getElementById('feedback-msg-wrap');
  var aStr=String(p.a),bStr=String(p.b);
  var wrongRows={};
  inputFields.forEach(function(el,i){if(!correctArr[i]&&rowInfo[i]>=0)wrongRows[rowInfo[i]]=true;});
  var hints=[];
  if(cfg.op==='vermenigvuldigen'){
    var steps=[];
    for(var bi=0;bi<bStr.length;bi++){
      var bd=parseInt(bStr[bStr.length-1-bi]),bPlace=Math.pow(10,bi);
      for(var ai=0;ai<aStr.length;ai++){
        var ad=parseInt(aStr[aStr.length-1-ai]),aPlace=Math.pow(10,ai);
        steps.push({formula:(ad*aPlace)+' \xd7 '+(bd*bPlace)+' = ?'});
      }
    }
    Object.keys(wrongRows).forEach(function(ri){
      ri=parseInt(ri);
      if(ri<steps.length)hints.push(steps[ri].formula);
    });
  } else {
    var maxPlaces=Math.max(aStr.length,bStr.length);
    var subtotals=[];
    for(var pos=maxPlaces-1;pos>=0;pos--){

      var dA=Math.floor(p.a/pv%10)*pv,dB=Math.floor(p.b/pv%10)*pv;
      var opS=p.op==='+'?'+':'-';
      subtotals.push({formula:dA+' '+opS+' '+dB+' = ?'});
    }
    Object.keys(wrongRows).forEach(function(ri){
      ri=parseInt(ri);
      if(ri<subtotals.length)hints.push(subtotals[ri].formula);
    });
  }

  if(hints.length>0&&msgWrap){

    });

  }
}

function getDeelsomLabel(p,rowIdx){
  var aStr=String(p.a),bStr=String(p.b);
  if(cfg.op==='vermenigvuldigen'){
    if(cfg.method==='kolomsgewijs'){
      var steps=[];
      for(var bi=0;bi<bStr.length;bi++){
        var bd=parseInt(bStr[bStr.length-1-bi]),bPlace=Math.pow(10,bi);
        for(var ai=0;ai<aStr.length;ai++){
          var ad=parseInt(aStr[aStr.length-1-ai]),aPlace=Math.pow(10,ai);
          steps.push((ad*aPlace)+' \xd7 '+(bd*bPlace)+' =');
        }
      }
      return rowIdx<steps.length?steps[rowIdx]:null;
    } else {
      var bRowIdx=Math.floor(rowIdx/aStr.length);
      var aDigIdx=rowIdx%aStr.length;
      var biPos=bStr.length-1-bRowIdx;
      var aiPos=aStr.length-1-aDigIdx;
      if(biPos<0||aiPos<0)return null;
      var bd=parseInt(bStr[biPos]);
      var ad=parseInt(aStr[aiPos]);
      if(isNaN(bd)||isNaN(ad))return null;
      return ad+' \xd7 '+bd+' =';
    }
  } else if(cfg.method==='cijferend'){
    var maxPlaces=Math.max(aStr.length,bStr.length);
    var subtotals=[];
    for(var pos=0;pos<maxPlaces;pos++){
      var pv=Math.pow(10,pos);
      var dA=Math.floor(p.a/pv%10)*pv,dB=Math.floor(p.b/pv%10)*pv;
      var opS=p.op==='+'?'+':'-';
      subtotals.push(dA+' '+opS+' '+dB+' =');
    }
    return rowIdx<subtotals.length?subtotals[rowIdx]:null;
  } else {
    var maxPlaces=Math.max(aStr.length,bStr.length);
    var subtotals=[];
    for(var pos=maxPlaces-1;pos>=0;pos--){
      var pv=Math.pow(10,pos);
      var dA=Math.floor(p.a/pv%10)*pv,dB=Math.floor(p.b/pv%10)*pv;
      var opS=p.op==='+'?'+':'-';
      subtotals.push(dA+' '+opS+' '+dB+' =');
    }
    return rowIdx<subtotals.length?subtotals[rowIdx]:null;
  }
}

function showVeelHulpForRow(ri){
  if(!cfg.veelHulp||!currentProblem)return;
  if(cfg.method==='cijferend')return;
  document.querySelectorAll('.veel-hulp-inline').forEach(function(e){e.parentNode&&e.parentNode.removeChild(e);});
  document.getElementById('hulp-area').innerHTML='';

  var label=null;

  // Kolomsgewijs vermenigvuldigen: optelrijen en einduitkomstrij
  if(cfg.method==='kolomsgewijs'&&cfg.op==='vermenigvuldigen'&&kolomVermMeta){
    var meta=kolomVermMeta;
    var stepsLen=meta.stepsLen;
    var colSums=meta.colSums;
    var numCols=meta.numCols;

    if(ri>=stepsLen&&ri<stepsLen+colSums.length){
      // Kolom-som-rij: lees ingevulde waarden van keersom-rijen in de juiste kolom
      var colSumIdx=ri-stepsLen;
      var cs=colSums[colSumIdx];
      var posExp=cs.pos;
      var posVal=Math.pow(10,posExp);
      var kolomIdx=meta.numCols-1-posExp;
      var delen=[];
      (meta.keersomInputs||[]).forEach(function(kr){
        var verwacht=Math.floor(kr.val/posVal)%10*posVal;
        if(verwacht===0)return; // nul-bijdrage overslaan
        var el=kr.inputsPerKolom[kolomIdx];
        if(el&&el.value.trim()!==''){
          // Gebruik ingevulde waarde, maar op de juiste positiewaarde
          var ingevuld=parseInt(el.value.trim())||0;
          delen.push(ingevuld*posVal);
        } else {
          delen.push(verwacht);
        }
      });
      if(delen.length===1){
        label=String(delen[0]);
      } else if(delen.length>1){
        label=delen.join(' + ')+' =';
      }
    } else if(ri===stepsLen+colSums.length){
      // Einduitkomst-rij: gebruik de waarden van de kolom-sommen
      var delenEind=[];
      colSums.forEach(function(cs){
        if(cs.val!==0)delenEind.push(cs.val);
      });
      // Lees de ingevulde waarden uit de kolom-som-invulvakjes
      var ingevuldDelenEind=[];
      colSums.forEach(function(cs,idx){
        var rijNr=stepsLen+idx;
        var rijInputs=[];
        for(var i=0;i<inputFields.length;i++){
          if(rowInfo[i]===rijNr&&inputFields[i])rijInputs.push(inputFields[i]);
        }
        var rijWaarde='';
        rijInputs.forEach(function(el){if(el.value.trim()!=='')rijWaarde+=el.value.trim();});
        var numVal=parseInt(rijWaarde);
        if(!isNaN(numVal)&&numVal!==0)ingevuldDelenEind.push(numVal);
      });
      // Gebruik altijd ingevulde waarden (ook als fout), anders verwachte waarden
      var bronDelen=ingevuldDelenEind.length>0?ingevuldDelenEind:delenEind;
      if(bronDelen.length===1){
        label=String(bronDelen[0]);
      } else if(bronDelen.length>1){
        label=bronDelen.join(' + ')+' =';
      }
    }
  }

  // Keersommen (normale fase): gebruik bestaande getDeelsomLabel
  if(label===null){
    label=getDeelsomLabel(currentProblem,ri);
  }

  if(!label)return;
  var targetInput=null;
  for(var i=0;i<inputFields.length;i++){if(rowInfo[i]===ri&&inputFields[i]){targetInput=inputFields[i];break;}}
  if(targetInput){
    var tr=targetInput.closest('tr');
    if(tr){
      var td=document.createElement('td');td.style.cssText='padding-left:16px;white-space:nowrap;vertical-align:middle;';
      var wrap=document.createElement('div');wrap.className='veel-hulp-inline';
      wrap.innerHTML='<span class="veel-hulp-inline-tekst">'+label+'</span>';
      td.appendChild(wrap);tr.appendChild(td);
    }
  }
  speakDeelsom(label.replace(/×/g,'keer').replace(/\+/g,'plus').replace(/-/g,'min').replace(/=/g,'is'));
}

function analyzeErrors(correctArr){
  var msg='Niet helemaal goed. ';
  var ansLen=String(Math.abs(currentProblem.answer)).length;
  var subWrong=correctArr.slice(0,correctArr.length-ansLen).some(function(c){return !c;});
  var ansWrong=correctArr.slice(-ansLen).some(function(c){return !c;});
  if(cfg.method==='kolomsgewijs'){
    if(subWrong&&!ansWrong)msg+='De tussenresultaten kloppen nog niet helemaal.';
    else if(ansWrong&&!subWrong)msg+='De tussenresultaten zijn goed, maar de eindoptelsom klopt niet.';
    else msg+='Controleer de tussenresultaten én de eindoptelsom.';
  } else {
    if(!correctArr[correctArr.length-1])msg+='De eenheden kloppen niet. Begin altijd rechts.';
    else if(!correctArr[0])msg+='Let op het onthouden of lenen.';
    else msg+='Controleer de rode vakjes zorgvuldig.';
  }
  return msg;
}

function kolombNaam(numCols, colIdx){

var rijNamen=['eerste','tweede','derde','vierde','vijfde','zesde','zevende','achtste','negende','tiende'];

function speakTabPositie(el){
  if(!cfg.autoVoorlezen||!currentProblem)return;
  // Zoek index van el in inputFields
  var idx=inputFields.indexOf(el);
  if(idx<0)return;
  var ri=rowInfo[idx];
  if(ri===undefined||ri<0)return;

  // Bepaal kolomnaam
  var numCols=0;
  // Zoek het aantal kolommen via de tabel in calc-area
  var tbl=document.querySelector('#calc-area table');
  if(tbl&&tbl.rows[0])numCols=tbl.rows[0].cells.length-1; // -1 voor label-col
  // Zoek kolompositie van el in zijn rij
  var td=el.closest('td');
  var tr=el.closest('tr');
  var colIdx=0;
  if(td&&tr){
    for(var ci=0;ci<tr.cells.length;ci++){if(tr.cells[ci]===td){colIdx=ci-1;break;}} // -1 voor label-col
  }
  var kolNaam=numCols>0?kolombNaam(numCols,colIdx):'?';

  // Bepaal rijnaam

// Globale leen-toetsenbord handler voor cijferend aftrekken
document.addEventListener('keydown',function(e){
  if(cfg.method!=='cijferend'||cfg.op!=='aftrekken')return;
  if(inputFields.indexOf(document.activeElement)<0)return;
  var k=e.key.toLowerCase();
  if(k!=='t'&&k!=='h'&&k!=='d')return;
  e.preventDefault();
  clearTimeout(leenToetsTimer);
  leenToetsBuffer+=k;
  var buffer=leenToetsBuffer;
  // T en H: wacht 1 seconde op tweede letter
  if(buffer.length===1&&(buffer==='t'||buffer==='h')){
    leenToetsTimer=setTimeout(function(){
      leenTrigger(leenToetsBuffer);
      leenToetsBuffer='';
    },1000);
  } else {
    // D alleen, of TD/HD combinatie
    leenTrigger(buffer);
    leenToetsBuffer='';
  }
});

function leenTrigger(naam){
  if(inputFields.indexOf(document.activeElement)<0)return;
  // Zoek cel met onclick op basis van kolomnaam (van rechts)
  var fromRightMap={e:0,t:1,h:2,d:3,td:4,hd:5};
  var fromRight=fromRightMap[naam];
  if(fromRight===undefined)return;
  // Bereken kolomindex van links: totaal aantal kolommen - 1 - fromRight
  // aStaticCells bevat de kolommen van getal A
  if(aStaticCells.length===0)return;
  var maxKol=Math.max.apply(null,aStaticCells.map(function(x){return x.col;}));
  var doelKol=maxKol-fromRight;
  var doel=aStaticCells.find(function(x){return x.col===doelKol&&x.wrap.onclick;});
  if(doel)doel.wrap.onclick();
}

}

function mkTbl(rows){
  var h='<table class="uitleg-table"><tbody>';
  rows.forEach(function(row){
    if(row==='hr'){h+='<tr class="uitleg-hr">'; for(var k=0;k<8;k++)h+='<td></td>';h+='</tr>';return;}
    h+='<tr>';
    row.forEach(function(cell){
      if(typeof cell==='object'){h+='<td class="'+(cell.cls||'')+'">'+cell.v+'</td>';}
      else{h+='<td>'+cell+'</td>';}
    });
    h+='</tr>';
  });
  return h+'</tbody></table>';
}

function errPattern(){
  var res=session.results.filter(function(r){return r&&!r.allCorrect;});if(!res.length)return null;
  if(cfg.method==='kolomsgewijs')return 'Gebruik de Uitleg-knop om nog eens te lezen hoe het werkt.';
  var rightErr=res.filter(function(r){return r.correct.length>0&&!r.correct[r.correct.length-1];}).length;
  if(rightErr>res.length/2)return 'De eenheden gaan vaak fout. Begin altijd rechts en werk naar links.';
  return 'Let goed op het onthouden of lenen bij de kolommen.';
}

  });
  rememberFields.forEach(function(el){if(el)el.disabled=false;});
  stepChecked=false;
  hideFeedback();
  clearEindHulp();
